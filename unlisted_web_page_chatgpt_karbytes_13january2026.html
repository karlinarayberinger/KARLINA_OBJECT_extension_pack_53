#!/usr/bin/env python3
"""
totp_demo.py

This program demonstrates how a 6-digit TOTP (Time-based One-Time Password)
is generated offline from:

  1) a shared secret key (the QR-code payload, usually Base32),
  2) the current time (UTC),
  3) a fixed time step (usually 30 seconds).

It prints the current code and the seconds remaining in the current 30s window.

NOTE:
- This is for understanding / personal verification.
- Do not hardcode real secrets in code you publish.
"""

import base64
import hashlib
import hmac
import struct
import time
from typing import Tuple


def _normalize_base32_secret(secret_b32: str) -> str:
    """
    Remove spaces and force uppercase, because Base32 secrets are often
    formatted with spaces for readability.
    """
    return secret_b32.replace(" ", "").strip().upper()


def _hotp(secret_bytes: bytes, counter: int, digits: int = 6) -> int:
    """
    HOTP(K, C) per RFC 4226:
    - K = secret key bytes
    - C = counter (integer)
    - returns an integer code with the requested number of digits
    """
    # 8-byte big-endian counter
    counter_bytes = struct.pack(">Q", counter)

    # HMAC-SHA1 is the classic default for TOTP
    hmac_digest = hmac.new(secret_bytes, counter_bytes, hashlib.sha1).digest()

    # Dynamic truncation (RFC 4226)
    offset = hmac_digest[-1] & 0x0F
    truncated = hmac_digest[offset : offset + 4]

    # 31-bit integer (most-significant bit cleared)
    code_int = struct.unpack(">I", truncated)[0] & 0x7FFFFFFF

    # Reduce to the requested number of digits
    return code_int % (10 ** digits)


def totp(
    secret_b32: str,
    timestep: int = 30,
    digits: int = 6,
    t0: int = 0,
    for_time: int | None = None,
) -> Tuple[str, int]:
    """
    TOTP per RFC 6238:
    TOTP = HOTP(K, (T - T0) / timestep)

    Returns:
      (code_str, seconds_remaining_in_window)
    """
    secret_b32 = _normalize_base32_secret(secret_b32)
    secret_bytes = base64.b32decode(secret_b32, casefold=True)

    now = int(time.time()) if for_time is None else int(for_time)
    counter = (now - t0) // timestep

    code_int = _hotp(secret_bytes, counter, digits=digits)
    code_str = str(code_int).zfill(digits)

    remaining = timestep - ((now - t0) % timestep)
    return code_str, remaining


def main() -> None:
    # Example Base32 secret for demonstration only.
    # Replace this with your own secret ONLY for local testing.
    demo_secret_b32 = "JBSWY3DPEHPK3PXP"

    code, remaining = totp(demo_secret_b32)
    print("TOTP demo (offline)")
    print("-------------------")
    print(f"Secret (Base32): {demo_secret_b32}")
    print(f"Current code:    {code}")
    print(f"Valid for:       {remaining} seconds")


if __name__ == "__main__":
    main()
